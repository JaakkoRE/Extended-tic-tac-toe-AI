package com.mycompany.tictactoeai.Methods;

/**
 *
 * @author Jaakko
 */
public class absoluteValue {
    /**
    * Method returns the absolute value
    * 
    * @param i given int 
    * 
    * @return int absolute value of i
    */
    public int absoluteValueOfInt(int i) {
        if (i < 0) {
            return -1 * i;
        } else {
            return i;
        }
    }
}
